<script>

    const WEB_APP_URL = "https://script.google.com/macros/s/AKfycbzUDnQ_MlcJ_etBMj7017WELSsAQUTtGn9Ma0hUB9xWbtGtvfQcLlKj3O70KIiEcqw/exec";

    

    // Experiment State

    let state = {

        participantId: "",

        framing: "",

        totalEarnings: 0,

        trialsCompleted: 0,

        redWhiteChoices: 0,

        blackGoldChoices: 0,

        isFlipping: false

    };


    const canvas = document.getElementById('task-canvas');

    const ctx = canvas.getContext('2d');


    function startExperiment() {

        const idInput = document.getElementById('p-id').value.trim();

        state.participantId = idInput || "USR-" + Math.random().toString(36).substring(2, 8).toUpperCase();

        

        // Assign Framing

        state.framing = Math.random() < 0.5 ? "Positive" : "Negative";

        

        // Set Instructions

        const instr = document.getElementById('instruction-text');

        if (state.framing === "Positive") {

            instr.innerText = "There is a 40% chance the coin will land on red and you will earn $15. If the coin lands on white, you will lose $7.";

        } else {

            instr.innerText = "There is a 60% chance the coin will land on white and you will lose $7. If the coin lands on red, you will earn $15.";

        }


        showScreen('screen-task');

        document.getElementById('earnings-box').style.display = 'block';

        drawCoins("initial");

    }


    function showScreen(id) {

        document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));

        document.getElementById(id).classList.add('active');

    }


    function drawCoins(resultType, colorA = null, colorB = null) {

        ctx.clearRect(0, 0, canvas.width, canvas.height);

        

        // Coin 1 (Red/White)

        drawCoinShape(200, 200, colorA || "#e74c3c", "Red / White");

        

        // Coin 2 (Black/Gold)

        drawCoinShape(600, 200, colorB || "#2c3e50", "Black / Gold");

    }


    function drawCoinShape(x, y, color, label) {

        ctx.beginPath();

        ctx.arc(x, y, 70, 0, Math.PI * 2);

        ctx.fillStyle = color;

        ctx.fill();

        ctx.strokeStyle = "#333";

        ctx.lineWidth = 5;

        ctx.stroke();

        ctx.closePath();

        

        ctx.fillStyle = (color === "#ffffff" || color === "#f1c40f") ? "#000" : "#fff";

        ctx.font = "bold 14px Arial";

        ctx.textAlign = "center";

        ctx.fillText(label, x, y + 5);

    }


    canvas.addEventListener('click', (e) => {

        if (state.isFlipping || state.trialsCompleted >= 20) return;


        const rect = canvas.getBoundingClientRect();

        const x = e.clientX - rect.left;

        const y = e.clientY - rect.top;


        // Check if Red/White Coin Clicked (Left)

        if (Math.sqrt((x-200)**2 + (y-200)**2) < 70) {

            handleFlip("RW");

        } 

        // Check if Black/Gold Coin Clicked (Right)

        else if (Math.sqrt((x-600)**2 + (y-600)**2) < 70) {

            handleFlip("BG");

        }

    });


    function handleFlip(choice) {

        state.isFlipping = true;

        let win = Math.random() < 0.40; // 40% chance to win ($15), 60% to lose ($7)

        

        let resultColor;

        if (choice === "RW") {

            state.redWhiteChoices++;

            resultColor = win ? "#e74c3c" : "#ffffff"; // Red or White

        } else {

            state.blackGoldChoices++;

            resultColor = win ? "#f1c40f" : "#2c3e50"; // Gold or Black

        }


        // Simple visual feedback

        let counter = 0;

        let anim = setInterval(() => {

            drawCoins("flipping", counter % 2 ? "#ccc" : "#999", counter % 2 ? "#999" : "#ccc");

            counter++;

            if (counter > 6) {

                clearInterval(anim);

                finishFlip(win, choice, resultColor);

            }

        }, 100);

    }


    function finishFlip(win, choice, resultColor) {

        if (win) state.totalEarnings += 15;

        else state.totalEarnings -= 7;


        state.trialsCompleted++;

        

        // Update Canvas with Result

        if (choice === "RW") drawCoins("result", resultColor, "#2c3e50");

        else drawCoins("result", "#e74c3c", resultColor);


        // Update UI

        document.getElementById('earnings-box').innerText = `Earnings: $${state.totalEarnings}`;

        document.getElementById('trial-count').innerText = state.trialsCompleted;


        state.isFlipping = false;


        if (state.trialsCompleted >= 20) {

            setTimeout(() => showScreen('screen-survey'), 1000);

        }

    }


    function submitFinalData(rating) {

        showScreen('screen-thanks');

        

        const payload = {

            Participant_ID: state.participantId,

            Framing: state.framing,

            Red_Chosen: state.redWhiteChoices, // "Times Red is Chosen" in your script

            Black_Chosen: state.blackGoldChoices, // "Times Black is Chosen" in your script

            Final_Earnings: state.totalEarnings,

            Rating: rating

        };


        // Send to Google Sheets

        fetch(WEB_APP_URL, {

            method: 'POST',

            mode: 'no-cors', // Critical for Apps Script Cross-Origin

            cache: 'no-cache',

            headers: { 'Content-Type': 'application/json' },

            body: JSON.stringify(payload)

        }).catch(err => console.error("Post Error:", err));

    }

</script>
